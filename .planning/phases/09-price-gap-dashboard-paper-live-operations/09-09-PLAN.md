       ```go
       // shouldLogNoFire returns true if this (symbol, reason) has not been
       // logged in the last priceGapNoFireLogCooldown window. Updates the map
       // on true. Flag-gated: caller MUST already have checked PriceGapDebugLog.
       func (t *Tracker) shouldLogNoFire(symbol, reason string, now time.Time) bool {
           if reason == "" {
               return false
           }
           key := symbol + "|" + reason
           t.debugLogMu.Lock()
           defer t.debugLogMu.Unlock()
           if last, ok := t.lastNoFireLogged[key]; ok && now.Sub(last) < priceGapNoFireLogCooldown {
               return false
           }
           t.lastNoFireLogged[key] = now
           return true
       }
       ```
