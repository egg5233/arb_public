    ```typescript
    type BreakerState = {
      enabled: boolean;
      pending_strike: number;
      sticky_until_ms: number;
      last_eval_pnl_usdt: number;
      last_eval_ts_ms: number;
      threshold_usdt: number;
      armed: boolean;
      tripped: boolean;
    };
    type BreakerTripRecord = { trip_ts_ms: number; trip_pnl_usdt: number; threshold_usdt: number; ramp_stage: number; paused_candidate_count: number; source: string; };

    export function usePgBreaker(token: string) {
      const [state, setState] = useState<BreakerState | null>(null);
      const [lastTrip, setLastTrip] = useState<BreakerTripRecord | null>(null);
      // 1. Initial fetch
      useEffect(() => {
        fetch('/api/pg/breaker/state', { headers: { Authorization: `Bearer ${token}` } })
          .then(r => r.json())
          .then(j => { if (j.ok) setState(j.data); });
      }, [token]);
      // 2. WS subscribe to pg.breaker.trip + pg.breaker.recover (extend existing WS hub subscription used by usePgRamp)
      // 3. Mutators
      async function recover(): Promise<{ ok: boolean; error?: string }> {
        const r = await fetch('/api/pg/breaker/recover', {
          method: 'POST',
          headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
          body: JSON.stringify({ confirmation_phrase: 'RECOVER' }),
        });
        return await r.json();
      }
      async function testFire(dryRun: boolean): Promise<{ ok: boolean; error?: string; data?: any }> {
        const r = await fetch('/api/pg/breaker/test-fire', {
          method: 'POST',
          headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
          body: JSON.stringify({ confirmation_phrase: 'TEST-FIRE', dry_run: dryRun }),
        });
        return await r.json();
      }
      return { state, lastTrip, recover, testFire };
    }
    ```

    ```tsx
    export function BreakerSubsection({ token }: { token: string }) {
      const { t } = useLocale();
      const { state, lastTrip, recover, testFire } = usePgBreaker(token);
      const [recoverOpen, setRecoverOpen] = useState(false);
      const [testFireOpen, setTestFireOpen] = useState(false);
      if (!state) return null;
      const badge = !state.enabled ? 'gray' : state.tripped ? 'red' : 'green';
      const badgeText = !state.enabled ? t('breakerDisabled') : state.tripped ? t('breakerTripped') : t('breakerArmed');
      return (
        <div className="breaker-subsection">
          <h4>{t('breakerStatus')}</h4>
          <Badge color={badge}>{badgeText}</Badge>
          <DataRow label={t('breaker24hPnL')} value={`${state.last_eval_pnl_usdt.toFixed(2)} USDT`} />
          <DataRow label={t('breakerThreshold')} value={`${state.threshold_usdt.toFixed(2)} USDT`} />
          {lastTrip && (
            <>
              <DataRow label={t('breakerLastTripTs')} value={fmtAsiaTaipei(lastTrip.trip_ts_ms)} />
              <DataRow label={t('breakerLastTripPnL')} value={`${lastTrip.trip_pnl_usdt.toFixed(2)} USDT`} />
              <DataRow label={t('breakerPausedCount')} value={String(lastTrip.paused_candidate_count)} />
            </>
          )}
          {state.tripped && (
            <button onClick={() => setRecoverOpen(true)}>{t('breakerRecoverButton')}</button>
          )}
          <button onClick={() => setTestFireOpen(true)} disabled={state.tripped}>
            {t('breakerTestFireButton')}
          </button>
          <BreakerConfirmModal
            open={recoverOpen}
            onClose={() => setRecoverOpen(false)}
            onConfirm={async () => { await recover(); setRecoverOpen(false); }}
            action="recover"
            magicPhrase="RECOVER"
            promptKey="breakerConfirmRecoverPrompt"
          />
          <BreakerConfirmModal
            open={testFireOpen}
            onClose={() => setTestFireOpen(false)}
            onConfirm={async ({ dryRun }) => { await testFire(!!dryRun); setTestFireOpen(false); }}
            action="test-fire"
            magicPhrase="TEST-FIRE"
            promptKey="breakerConfirmTestFirePrompt"
          />
        </div>
      );
    }
    ```
