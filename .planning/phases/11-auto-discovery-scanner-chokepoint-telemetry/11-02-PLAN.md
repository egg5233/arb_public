    func (c *Config) SaveJSONWithBakRing() error {
        // Caller MUST hold c.mu (same contract as SaveJSON; see config.go:778-781).
        // 1. Resolve config path via existing CONFIG_FILE env / cwd lookup helper.
        path, err := c.resolveConfigPath() // reuse existing helper at ~config.go:1583
        if err != nil { return err }

        // 2. Build the JSON bytes via the existing SaveJSON marshalling path
        //    (factor out the bytes-build into a helper if needed; preserve
        //    keepNonZero tripwire + exchange-secret protection).
        data, err := c.marshalForSave(nil)
        if err != nil { return err }

        // 3. Write tmp file
        tmpPath := path + ".tmp"
        f, err := os.OpenFile(tmpPath, os.O_WRONLY|os.O_CREATE|os.O_TRUNC, 0644)
        if err != nil { return err }
        if _, err := f.Write(data); err != nil { f.Close(); return err }
        if err := f.Sync(); err != nil { f.Close(); return err } // fsync
        if err := f.Close(); err != nil { return err }

        // 4. Atomic rename — testable via c.renameFunc
        rename := c.renameFunc
        if rename == nil { rename = os.Rename }
        if err := rename(tmpPath, path); err != nil {
            os.Remove(tmpPath)
            return err
        }

        // 5. Write timestamped .bak.{unix_ts} (a copy of post-rename content)
        bakPath := fmt.Sprintf("%s.bak.%d", path, time.Now().Unix())
        if err := os.WriteFile(bakPath, data, 0644); err != nil {
            // non-fatal — primary save already succeeded
            log.Printf("[config] bak write failed: %v", err)
        }

        // 6. Prune ring to newest 5
        c.pruneBakRing(path)
        return nil
    }

    type Registry struct {
        cfg     *config.Config
        db      database.RegistryAuditWriter // narrow interface — see below
        log     *utils.Logger
        nowFunc func() time.Time // injectable for tests

    // RegistryAuditWriter is a narrow surface — only what Registry needs from

    func NewRegistry(cfg *config.Config, db RegistryAuditWriter, log *utils.Logger) *Registry {
        return &Registry{cfg: cfg, db: db, log: log, nowFunc: time.Now}

    var _ RegistryReader = (*Registry)(nil)

    func (r *Registry) Get(idx int) (models.PriceGapCandidate, bool) {
        r.cfg.RLock()
        defer r.cfg.RUnlock()
        if idx < 0 || idx >= len(r.cfg.PriceGapCandidates) {
            return models.PriceGapCandidate{}, false
        }
        return r.cfg.PriceGapCandidates[idx], true
    }

    func (r *Registry) List() []models.PriceGapCandidate {
        r.cfg.RLock()
        defer r.cfg.RUnlock()
        out := make([]models.PriceGapCandidate, len(r.cfg.PriceGapCandidates))
        copy(out, r.cfg.PriceGapCandidates)
        return out
    }

    func (r *Registry) Add(ctx context.Context, source string, c models.PriceGapCandidate) error {
        r.cfg.Lock()
        defer r.cfg.Unlock()
        // 1. Reload from disk (Pitfall 2 belt-and-suspenders for cross-process)

        // 2. Cap check (use cfg.PriceGapMaxCandidates from Plan 01 schema)
        if len(r.cfg.PriceGapCandidates) >= r.cfg.PriceGapMaxCandidates {
            return ErrCapExceeded
        }
        // 3. Dedupe by (Symbol, LongExch, ShortExch, Direction)
        for _, existing := range r.cfg.PriceGapCandidates {
            if c.Symbol == existing.Symbol &&
               c.LongExch == existing.LongExch &&
               c.ShortExch == existing.ShortExch &&
               c.Direction == existing.Direction {
                return ErrDuplicateCandidate
            }
        }
        // 4. Append

        r.cfg.PriceGapCandidates = append(r.cfg.PriceGapCandidates, c)
        // 5. Persist via SaveJSONWithBakRing (Task 1)
        if err := r.cfg.SaveJSONWithBakRing(); err != nil {
            // rollback in-memory mutation
            r.cfg.PriceGapCandidates = r.cfg.PriceGapCandidates[:beforeCount]
            return fmt.Errorf("registry: save: %w", err)
        }
        // 6. Audit trail
        r.writeAuditLocked(ctx, source, "add", beforeCount, beforeCount+1)
        return nil
    }
