    ```go
    package pricegaptrader

    import "github.com/<repo>/internal/config"

    // Sizer caps Strategy 4 notional sizing per the live-capital ramp.
    // Defense in depth: sizer caps at the sizing call site, risk_gate Gate 6
    // independently rejects over-budget proposals (some other code path could
    // bypass the sizer; gate is the second layer).
    type Sizer struct {
        cfg *config.Config
    }

    func NewSizer(cfg *config.Config) *Sizer { return &Sizer{cfg: cfg} }

    // Cap returns min(notional, stage_size, hard_ceiling).
    // When PriceGapLiveCapital=false, returns notional unchanged (paper-mode pass-through).
    // When stage is out of [1,3] range, returns 0 (fail-closed).
    func (s *Sizer) Cap(notional float64, stage int) float64 {
        if s.cfg == nil || !s.cfg.PriceGapLiveCapital {
            return notional
        }
        stageSize := s.stageSize(stage)
        if stageSize <= 0 {
            return 0
        }
        cap := stageSize
        if s.cfg.PriceGapHardCeilingUSDT > 0 && s.cfg.PriceGapHardCeilingUSDT < cap {
            cap = s.cfg.PriceGapHardCeilingUSDT
        }
        if notional < cap {
            return notional
        }
        return cap
    }

    func (s *Sizer) stageSize(stage int) float64 {
        switch stage {
        case 1:
            return s.cfg.PriceGapStage1SizeUSDT
        case 2:
            return s.cfg.PriceGapStage2SizeUSDT
        case 3:
            return s.cfg.PriceGapStage3SizeUSDT
        default:
            return 0
        }
    }
    ```

    ```go
        notifier RampNotifier
        nowFn    func() time.Time

        rc := &RampController{store: store, notifier: notifier, cfg: cfg, log: log, nowFn: nowFn}

        return rc
    }

    func (rc *RampController) bootstrapState() {
        s, exists, err := rc.store.LoadPriceGapRampState()
        if err != nil {
            rc.log.Errorf("pricegap ramp: load state failed: %v", err); return
        }
        if !exists || s.CurrentStage < 1 {
            // T-14-12: legacy/empty state → bootstrap to stage 1
            s = models.RampState{CurrentStage: 1, CleanDayCounter: 0}
            if err := rc.store.SavePriceGapRampState(s); err != nil {
                rc.log.Warnf("pricegap ramp: bootstrap save failed: %v", err)
            }
        }
        rc.state, rc.loaded = s, true
    }

    func (rc *RampController) Snapshot() models.RampState {
        rc.mu.Lock(); defer rc.mu.Unlock()
        return rc.state
    }

    // Eval drives ramp transitions from a single day's reconcile record.
    // Idempotent: same date evaluated twice has no extra effect (LastEvalTs guard).
    func (rc *RampController) Eval(ctx context.Context, date string, record DailyReconcileRecord) {
        rc.mu.Lock(); defer rc.mu.Unlock()
        // Idempotency: same-date reentry → no-op (D-04 spirit applied to ramp)
        evalDate, _ := time.Parse("2006-01-02", date)
        if !rc.state.LastEvalTs.IsZero() && rc.state.LastEvalTs.Format("2006-01-02") == date {
            return
        }
        prior := rc.state.CurrentStage
        // D-05: clean day = realized PnL ≥ 0 AND ≥1 closed position
        // No-activity day = PositionsClosed == 0 → HOLD counter (no change)
        if record.Totals.PositionsClosed == 0 {
            rc.state.LastEvalTs = evalDate
            _ = rc.persistLocked("evaluate", prior, prior, "no_activity_hold")
            return
        }
        if record.Totals.NetClean {
            rc.state.CleanDayCounter++
            // Promotion threshold reached?
            if rc.state.CurrentStage < 3 && rc.state.CleanDayCounter >= rc.cfg.PriceGapCleanDaysToPromote {
                rc.state.CurrentStage++
                rc.state.CleanDayCounter = 0
            }
        } else {
            // Loss day → asymmetric ratchet: counter=0 + demote one stage
            rc.state.CleanDayCounter = 0
            rc.state.LastLossDayTs = evalDate
            if rc.state.CurrentStage > 1 {
                rc.state.CurrentStage--
                rc.state.DemoteCount++
                if rc.notifier != nil {
                    rc.notifier.NotifyPriceGapRampDemote(prior, rc.state.CurrentStage,
                        fmt.Sprintf("loss_day pnl=%.2f", record.Totals.RealizedPnLUSDT))
                }
            }
        }
        rc.state.LastEvalTs = evalDate
        reason := "clean_day"
        if !record.Totals.NetClean { reason = "loss_day" }
        _ = rc.persistLocked("evaluate", prior, rc.state.CurrentStage, reason)
    }

    func (rc *RampController) ForcePromote(operator, reason string) error {
        rc.mu.Lock(); defer rc.mu.Unlock()
        prior := rc.state.CurrentStage
        if prior >= 3 { return fmt.Errorf("already at max stage 3") }
        rc.state.CurrentStage++
        // D-15 #3: counter PRESERVED — operator override is intentional
        return rc.persistLocked("force_promote", prior, rc.state.CurrentStage, fmt.Sprintf("op=%s reason=%s", operator, reason))
    }

    func (rc *RampController) ForceDemote(operator, reason string) error {
        rc.mu.Lock(); defer rc.mu.Unlock()
        prior := rc.state.CurrentStage
        if prior <= 1 { return fmt.Errorf("already at min stage 1") }
        rc.state.CurrentStage--
        rc.state.CleanDayCounter = 0  // D-15 #4 — demote zeroes counter (matches asymmetric ratchet)
        rc.state.DemoteCount++
        return rc.persistLocked("force_demote", prior, rc.state.CurrentStage, fmt.Sprintf("op=%s reason=%s", operator, reason))
    }

    func (rc *RampController) Reset(operator, reason string) error {
        rc.mu.Lock(); defer rc.mu.Unlock()
        prior := rc.state.CurrentStage
        rc.state = models.RampState{CurrentStage: 1, CleanDayCounter: 0}
        return rc.persistLocked("reset", prior, 1, fmt.Sprintf("op=%s reason=%s", operator, reason))
    }

    func (rc *RampController) persistLocked(action string, prior, next int, reason string) error {
        if err := rc.store.SavePriceGapRampState(rc.state); err != nil {
            rc.log.Errorf("pricegap ramp: SavePriceGapRampState failed: %v", err)
            return err
        }
        ev := models.RampEvent{
            Action: action, PriorStage: prior, NewStage: next,
            Operator: "daemon", Timestamp: rc.nowFn().UTC(), Reason: reason,
        }
        if err := rc.store.AppendPriceGapRampEvent(ev); err != nil {
            rc.log.Warnf("pricegap ramp: AppendPriceGapRampEvent failed: %v", err)
        }
        return nil
    }
    ```
