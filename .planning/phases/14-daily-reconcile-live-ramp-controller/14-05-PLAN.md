    ```go
    package api

    import (
        "encoding/json"

    // GET /api/pg/ramp
    func (s *Server) handlePgRampState(w http.ResponseWriter, r *http.Request) {
        if s.pgRamp == nil {
            writeJSON(w, http.StatusServiceUnavailable, Response{Ok: false, Error: "ramp controller not configured"})
            return
        }
        snap := s.pgRamp.Snapshot()
        // Server-authoritative live-capital flag (NOT from client config cache)
        liveCapital := s.cfg != nil && s.cfg.PriceGapLiveCapital
        var stage1, stage2, stage3, ceiling float64
        if s.cfg != nil {
            stage1 = s.cfg.PriceGapStage1SizeUSDT
            stage2 = s.cfg.PriceGapStage2SizeUSDT
            stage3 = s.cfg.PriceGapStage3SizeUSDT
            ceiling = s.cfg.PriceGapHardCeilingUSDT
        }
        writeJSON(w, http.StatusOK, Response{Ok: true, Data: map[string]any{
            "current_stage":      snap.CurrentStage,
            "clean_day_counter":  snap.CleanDayCounter,
            "last_eval_ts":       snap.LastEvalTs,
            "last_loss_day_ts":   snap.LastLossDayTs,
            "demote_count":       snap.DemoteCount,
            "live_capital":       liveCapital,
            "stage_1_size_usdt":  stage1,
            "stage_2_size_usdt":  stage2,
            "stage_3_size_usdt":  stage3,
            "hard_ceiling_usdt":  ceiling,
            "clean_days_to_promote": s.cfg.PriceGapCleanDaysToPromote,
        }})
    }

    var pgReconcileDateRegex = regexp.MustCompile(`^\d{4}-\d{2}-\d{2}$`)

    // GET /api/pg/reconcile/{date}
    func (s *Server) handlePgReconcileDay(w http.ResponseWriter, r *http.Request) {
        if s.pgReconciler == nil {
            writeJSON(w, http.StatusServiceUnavailable, Response{Ok: false, Error: "reconciler not configured"})
            return
        }
        date := r.PathValue("date")
        if !pgReconcileDateRegex.MatchString(date) {
            writeJSON(w, http.StatusBadRequest, Response{Ok: false, Error: "invalid date format (expect YYYY-MM-DD)"})
            return
        }
        rec, exists, err := s.pgReconciler.LoadRecord(r.Context(), date)
        if err != nil {
            writeJSON(w, http.StatusInternalServerError, Response{Ok: false, Error: err.Error()})
            return
        }
        if !exists {
            writeJSON(w, http.StatusNotFound, Response{Ok: false, Error: "reconcile not found for date"})
            return
        }
        writeJSON(w, http.StatusOK, Response{Ok: true, Data: rec})
    }
    ```

    **B. Create RampReconcileSection component** — `web/src/components/Ramp/RampReconcileSection.tsx` (new directory + file):

    ```tsx
    // READ-ONLY widget (D-14). Force-promote/demote/reset live in pg-admin only this phase.
    // Phase 16 (PG-OPS-09) will absorb this widget into the new top-level Price-Gap tab.
    // DO NOT add buttons that POST anywhere — that's Phase 16 territory.

    import { useEffect, useState } from 'react';
    import { useLocale } from '../../i18n/useLocale';

    interface RampState {
      current_stage: number;
      clean_day_counter: number;
      last_eval_ts: string;
      last_loss_day_ts: string;
      demote_count: number;
      live_capital: boolean;
      stage_1_size_usdt: number;
      stage_2_size_usdt: number;
      stage_3_size_usdt: number;
      hard_ceiling_usdt: number;
      clean_days_to_promote: number;
    }

    interface DailyTotals {
      realized_pnl_usdt: number;
      positions_closed: number;
      wins: number;
      losses: number;
    }

    interface DailyAnomalies {
      high_slippage_count: number;
      missing_close_ts_count: number;
      flagged_ids: string[];
    }

    interface DailyRecord {
      date: string;
      computed_at: string;
      totals: DailyTotals;
      anomalies: DailyAnomalies;
    }

    export function RampReconcileSection() {
      const { t } = useLocale();
      const [ramp, setRamp] = useState<RampState | null>(null);
      const [reconcile, setReconcile] = useState<DailyRecord | null>(null);

      useEffect(() => {
        const token = localStorage.getItem('arb_token');
        const headers = token ? { Authorization: `Bearer ${token}` } : {};
        fetch('/api/pg/ramp', { headers })
          .then((r) => r.json())
          .then((j) => { if (j.ok) setRamp(j.data); });
        // Yesterday UTC
        const y = new Date();
        y.setUTCDate(y.getUTCDate() - 1);
        const dateStr = y.toISOString().slice(0, 10);
        fetch(`/api/pg/reconcile/${dateStr}`, { headers })
          .then((r) => r.json())
          .then((j) => { if (j.ok) setReconcile(j.data); });
      }, []);

      if (!ramp) {
        return <div className="p-4 text-gray-500">Loading ramp state…</div>;
      }

      const badgeClass = ramp.live_capital
        ? 'inline-block px-3 py-1 rounded font-bold bg-red-600 text-white'
        : 'inline-block px-3 py-1 rounded font-bold bg-gray-400 text-white';
      const badgeLabel = ramp.live_capital
        ? t('pricegap.ramp.liveCapitalOn')
        : t('pricegap.ramp.liveCapitalOff');

      const stageSize = [ramp.stage_1_size_usdt, ramp.stage_2_size_usdt, ramp.stage_3_size_usdt][ramp.current_stage - 1] ?? 0;

      return (
        <section className="my-6 p-4 border border-gray-300 rounded-lg" data-test="ramp-reconcile-section">
          <div className="flex items-center justify-between mb-4">
            <h2 className="text-lg font-bold">{t('pricegap.ramp.title')}</h2>
            <span className={badgeClass} data-test="live-capital-badge">{badgeLabel}</span>
          </div>
          <div className="grid grid-cols-2 md:grid-cols-3 gap-3 text-sm">
            <div><span className="text-gray-500">{t('pricegap.ramp.stage')}:</span> <strong>{ramp.current_stage}/3</strong> (${stageSize.toFixed(0)}/leg)</div>
            <div><span className="text-gray-500">{t('pricegap.ramp.cleanDayCounter')}:</span> <strong>{ramp.clean_day_counter}/{ramp.clean_days_to_promote}</strong></div>
            <div><span className="text-gray-500">{t('pricegap.ramp.demoteCount')}:</span> {ramp.demote_count}</div>
            <div><span className="text-gray-500">{t('pricegap.ramp.lastLossDay')}:</span> {ramp.last_loss_day_ts.startsWith('0001') ? '—' : ramp.last_loss_day_ts.slice(0, 10)}</div>
            <div><span className="text-gray-500">Hard Ceiling:</span> ${ramp.hard_ceiling_usdt.toFixed(0)}/leg</div>
          </div>

          {reconcile ? (
            <div className="mt-6 pt-4 border-t border-gray-200">
              <h3 className="text-md font-semibold mb-2">{t('pricegap.reconcile.title')} ({reconcile.date})</h3>
              <div className="grid grid-cols-2 md:grid-cols-3 gap-3 text-sm">
                <div><span className="text-gray-500">{t('pricegap.reconcile.totalPnl')}:</span> <strong className={reconcile.totals.realized_pnl_usdt >= 0 ? 'text-green-700' : 'text-red-700'}>${reconcile.totals.realized_pnl_usdt.toFixed(2)}</strong></div>
                <div><span className="text-gray-500">{t('pricegap.reconcile.positionsClosed')}:</span> {reconcile.totals.positions_closed}</div>
                <div><span className="text-gray-500">{t('pricegap.reconcile.winLossSplit')}:</span> {reconcile.totals.wins}W / {reconcile.totals.losses}L</div>
              </div>
              <div className="mt-3">
                <span className="text-gray-500 text-sm">{t('pricegap.reconcile.anomalies')}:</span>{' '}
                {reconcile.anomalies.flagged_ids.length === 0
                  ? <span className="text-green-700 text-sm">{t('pricegap.reconcile.noAnomalies')}</span>
                  : <span className="text-orange-700 text-sm" data-test="anomaly-list">
                      {reconcile.anomalies.high_slippage_count} high-slip + {reconcile.anomalies.missing_close_ts_count} missing-ts: {reconcile.anomalies.flagged_ids.join(', ')}
                    </span>}
              </div>
            </div>
          ) : (
            <div className="mt-4 text-gray-500 text-sm">No reconcile data for yesterday yet.</div>
          )}
        </section>
      );
    }
    ```
