    ```go
    // reconcileLoop fires UTC 00:30 daily (D-16 hardcoded).
    // Imitates scanLoop pattern (RESEARCH Q6).
    func (t *Tracker) reconcileLoop() {
        defer t.wg.Done()
        if t.reconciler == nil || t.ramp == nil {
            t.log.Info("pricegap reconcile daemon disabled (no Reconciler/Ramp wired)")
            return
        }
        runOnce := func(date string) {
            ctx, cancel := context.WithTimeout(context.Background(), 10*time.Minute)
            defer cancel()
            if err := t.reconciler.RunForDate(ctx, date); err != nil {
                t.log.Errorf("pricegap: reconcile %s daemon-run failed: %v", date, err)
                return // NotifyPriceGapReconcileFailure already dispatched inside RunForDate triple-fail path
            }
            rec, exists, err := t.reconciler.LoadRecord(ctx, date)
            if err != nil || !exists {
                t.log.Errorf("pricegap: reconcile %s record not loadable post-run: %v", date, err)
                return
            }
            t.ramp.Eval(ctx, date, rec)
            if t.notifier != nil {
                t.notifier.NotifyPriceGapDailyDigest(date, rec, t.ramp.Snapshot())
            }
        }

        // Boot-time catchup (Q14): if past 01:00 UTC and yesterday's reconcile missing → run-immediately.
        nowUTC := time.Now().UTC()
        if nowUTC.Hour() >= 1 {
            yesterday := nowUTC.AddDate(0, 0, -1).Format("2006-01-02")
            if _, exists, _ := t.reconciler.LoadRecord(context.Background(), yesterday); !exists {
                t.log.Infof("pricegap: boot-catchup running reconcile for %s", yesterday)
                runOnce(yesterday)
            }
        }

        // Settle into normal cadence — fire every UTC 00:30
        for {
            next := nextUTCFireTime(time.Now())
            select {
            case <-t.stopCh:
                return
            case <-time.After(time.Until(next)):
                date := time.Now().UTC().AddDate(0, 0, -1).Format("2006-01-02")
                runOnce(date)
            }
        }
    }

    // nextUTCFireTime returns the next UTC 00:30 instant strictly after now.
    func nextUTCFireTime(now time.Time) time.Time {
        u := now.UTC()
        target := time.Date(u.Year(), u.Month(), u.Day(), 0, 30, 0, 0, time.UTC)
        if !target.After(u) {
            target = target.AddDate(0, 0, 1)
        }
        return target
    }
    ```

    ```go
    func cmdReconcileRun(args []string, deps Dependencies, stdout, stderr io.Writer) int {
        date, err := parseReconcileDateFlag(args, "run")
        if err != nil {
            fmt.Fprintf(stderr, "error: %v\n", err); return 2
        }
        if deps.Reconciler == nil {
            fmt.Fprintln(stderr, "error: reconciler not wired (deps.Reconciler is nil)"); return 2
        }
        ctx, cancel := context.WithTimeout(context.Background(), 10*time.Minute)
        defer cancel()
        // Skip-if-exists per D-04 (idempotent re-run still safe but ops want fast feedback)
        if _, exists, _ := deps.Reconciler.LoadRecord(ctx, date); exists {
            // Re-run anyway — D-04 byte-identical guarantee. Just inform.
            fmt.Fprintf(stdout, "note: %s already reconciled — re-running to verify byte-identical output\n", date)
        }
        if err := deps.Reconciler.RunForDate(ctx, date); err != nil {
            fmt.Fprintf(stderr, "reconcile %s FAILED: %v\n", date, err); return 2
        }
        fmt.Fprintf(stdout, "reconcile %s OK\n", date); return 0
    }

    // cmdReconcileShow: pg-admin reconcile show --date=YYYY-MM-DD
    // Output: JSON-indent (mirror cmdCandidatesList pattern at main.go:1442)
    func cmdReconcileShow(args []string, deps Dependencies, stdout, stderr io.Writer) int {
        date, err := parseReconcileDateFlag(args, "show")
        if err != nil { fmt.Fprintf(stderr, "error: %v\n", err); return 2 }
        ctx := context.Background()
        rec, exists, err := deps.Reconciler.LoadRecord(ctx, date)
        if err != nil { fmt.Fprintf(stderr, "error: %v\n", err); return 2 }
        if !exists { fmt.Fprintf(stderr, "no record for %s\n", date); return 1 }
        enc := json.NewEncoder(stdout); enc.SetIndent("", "  ")
        if err := enc.Encode(rec); err != nil {
            fmt.Fprintf(stderr, "encode error: %v\n", err); return 2
        }
        return 0
    }

    var dateRegex = regexp.MustCompile(`^\d{4}-\d{2}-\d{2}$`)

    func parseReconcileDateFlag(args []string, sub string) (string, error) {
        for _, a := range args {
            if len(a) > 7 && a[:7] == "--date=" {
                d := a[7:]
                if !dateRegex.MatchString(d) {
                    return "", fmt.Errorf("invalid --date=%s (expect YYYY-MM-DD)", d)
                }
                if t, e := time.Parse("2006-01-02", d); e != nil || t.After(time.Now().UTC().AddDate(0, 0, 1)) {
                    return "", fmt.Errorf("invalid or future --date=%s", d)
                }
                return d, nil
            }
        }
        return "", fmt.Errorf("reconcile %s requires --date=YYYY-MM-DD", sub)
    }
    ```
