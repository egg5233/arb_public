```go
package api

import (
    "net/http"
    "strconv"
    "time"
    "arb/internal/analytics"
)

// handleGetAnalyticsPnLHistory returns time-series PnL snapshots.
// Query params: from (unix seconds), to (unix seconds), strategy (perp|spot|all)
func (s *Server) handleGetAnalyticsPnLHistory(w http.ResponseWriter, r *http.Request) {
    if r.Method != http.MethodGet {
        http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
        return
    }
    if s.analyticsStore == nil {
        writeJSON(w, http.StatusServiceUnavailable, Response{Error: "analytics not available"})
        return
    }

    from, _ := strconv.ParseInt(r.URL.Query().Get("from"), 10, 64)
    to, _ := strconv.ParseInt(r.URL.Query().Get("to"), 10, 64)
    strategy := r.URL.Query().Get("strategy")

    if to == 0 {
        to = time.Now().Unix()
    }
    if from == 0 {
        from = to - 30*86400 // default 30 days
    }
    if strategy == "" {
        strategy = "all"
    }

    snapshots, err := s.analyticsStore.GetPnLHistory(from, to, strategy)
    if err != nil {
        s.log.Error("analytics pnl-history: %v", err)
        writeJSON(w, http.StatusInternalServerError, Response{Error: "analytics query failed"})
        return
    }
    writeJSON(w, http.StatusOK, Response{OK: true, Data: snapshots})
}

// handleGetAnalyticsSummary returns strategy summaries and exchange metrics.
// Query params: from (unix seconds), to (unix seconds)
func (s *Server) handleGetAnalyticsSummary(w http.ResponseWriter, r *http.Request) {
    if r.Method != http.MethodGet {
        http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
        return
    }

    // Get perp history from Redis
    perps, err := s.db.GetHistory(500)
    if err != nil {
        s.log.Error("analytics summary perp history: %v", err)
        writeJSON(w, http.StatusInternalServerError, Response{Error: "failed to fetch perp history"})
        return
    }
    // Get spot history from Redis
    spots, err := s.db.GetSpotHistory(500)
    if err != nil {
        s.log.Error("analytics summary spot history: %v", err)
        writeJSON(w, http.StatusInternalServerError, Response{Error: "failed to fetch spot history"})
        return
    }

    // Apply time filter if provided
    from, _ := strconv.ParseInt(r.URL.Query().Get("from"), 10, 64)
    to, _ := strconv.ParseInt(r.URL.Query().Get("to"), 10, 64)
    if to == 0 {
        to = time.Now().Unix()
    }
    if from == 0 {
        from = to - 30*86400
    }
    fromT := time.Unix(from, 0)
    toT := time.Unix(to, 0)

    // Filter perps by close time (UpdatedAt for closed positions)
    var filteredPerps []*models.ArbitragePosition
    for _, p := range perps {
        if p.UpdatedAt.After(fromT) && p.UpdatedAt.Before(toT) {
            filteredPerps = append(filteredPerps, p)
        }
    }
    var filteredSpots []*models.SpotFuturesPosition
    for _, p := range spots {
        if p.UpdatedAt.After(fromT) && p.UpdatedAt.Before(toT) {
            filteredSpots = append(filteredSpots, p)
        }
    }

    type summaryResponse struct {
        Strategies      []analytics.StrategySummary `json:"strategies"`
        ExchangeMetrics []analytics.ExchangeMetric  `json:"exchange_metrics"`
    }

    resp := summaryResponse{
        Strategies:      analytics.ComputeStrategySummary(filteredPerps, filteredSpots),
        ExchangeMetrics: analytics.ComputeExchangeMetrics(filteredPerps, filteredSpots),
    }
    writeJSON(w, http.StatusOK, Response{OK: true, Data: resp})
}
```

```go
package analytics

import (
    "arb/internal/models"
    "arb/pkg/utils"
    "time"
)

// SnapshotWriter receives position-close events on a buffered channel
// and writes PnL snapshots to the SQLite store in a background goroutine.
type SnapshotWriter struct {
    store  *Store
    ch     chan snapshotEvent
    log    *utils.Logger
    stopCh chan struct{}
}

type snapshotEvent struct {
    Strategy      string
    Exchange      string
    PnL           float64
    IsWin         bool
    FundingTotal  float64
    FeesTotal     float64
}

func NewSnapshotWriter(store *Store) *SnapshotWriter
func (sw *SnapshotWriter) Start()                      // goroutine that reads from ch and writes snapshots
func (sw *SnapshotWriter) Stop()                       // signal stop
func (sw *SnapshotWriter) RecordPerpClose(pos *models.ArbitragePosition)          // send event to channel
func (sw *SnapshotWriter) RecordSpotClose(pos *models.SpotFuturesPosition)        // send event to channel
```

The `Start()` goroutine reads from the buffered channel (size 100), builds a `PnLSnapshot` with `Timestamp: time.Now().Unix()`, and calls `store.WritePnLSnapshot()`. Non-blocking -- if channel full, log warning and drop (analytics data is best-effort, must not affect trading).

The `RecordPerpClose` method sends a snapshotEvent with:
- Strategy: "perp"
- Exchange: "" (aggregate -- perp spans two exchanges)
- PnL: pos.RealizedPnL
- IsWin: pos.RealizedPnL > 0
- FundingTotal: pos.FundingCollected
- FeesTotal: pos.EntryFees + pos.ExitFees

The `RecordSpotClose` method sends a snapshotEvent with:
- Strategy: "spot"
- Exchange: pos.Exchange
- PnL: pos.RealizedPnL
- IsWin: pos.RealizedPnL > 0
- FundingTotal: pos.FundingCollected
- FeesTotal: pos.EntryFees + pos.ExitFees

**2c. Capture BBO snapshot before PlaceOrder calls** in `internal/engine/engine.go`:

Find the main entry execution path where both legs are placed (the `executeArbitrage` or similar function that calls `PlaceOrder` for long and short legs). Before each `PlaceOrder`:

3. Store cumulative slippage on the position:

   ```go
   pos.Slippage = longSlippage + shortSlippage
   ```

**Where to hook this:** Search for PlaceOrder calls in `engine.go` that are part of entry execution. The main entry path is in the execution flow that follows `executeArbitrage`. Look for where `LongEntry` and `ShortEntry` are set on the position after fills.

If the PlaceOrder call sites are too complex to modify safely (many code paths, error handling), a simpler approach: capture BBO right before the position is saved/persisted during entry, and store the delta between entry price and BBO as slippage. This is less precise but zero-risk to the trading path.

**Simpler safe approach:** In the entry path, right after both legs fill and before the position is saved to Redis, add:
```go
// BBO slippage estimation (best-effort, non-blocking)
var slippage float64
if longBBO, ok := longExch.GetBBO(pos.Symbol); ok && pos.LongEntry > 0 {
    slippage += pos.LongEntry - longBBO.Ask // buy: positive = worse than ask
}
if shortBBO, ok := shortExch.GetBBO(pos.Symbol); ok && pos.ShortEntry > 0 {
    slippage += shortBBO.Bid - pos.ShortEntry // sell: positive = worse than bid
}
pos.Slippage = slippage
```

This captures an approximate BBO delta at position creation time. The BBO snapshot from the WebSocket price store is near-real-time and sufficient for analytics purposes.
