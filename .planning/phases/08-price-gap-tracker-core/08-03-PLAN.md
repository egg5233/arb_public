    ```go
    package pricegaptrader

    import (
        "fmt"
        "math"
        "time"

        "arb/internal/models"
        "arb/pkg/exchange"
    )

    // candidateBars — per-candidate in-memory rolling bar state.
    // Key in Tracker.bars map = PriceGapCandidate.ID().
    type candidateBars struct {
        ring        barRing
        lastSampleAt time.Time
    }

    // barRing — 4-slot circular buffer of 1m spread closes (D-07, PG-01).
    type barRing struct {
        closes  [4]float64 // spread bps
        valid   [4]bool
        idx     int
        lastMin int64 // unix minute of most recent completed bar
    }

    // push records a new 1m close; returns true if the minute was new (actually stored).
    func (b *barRing) push(unixMinute int64, spreadBps float64) bool {
        if unixMinute == b.lastMin { return false }
        b.closes[b.idx] = spreadBps
        b.valid[b.idx] = true
        b.idx = (b.idx + 1) % 4
        b.lastMin = unixMinute
        return true
    }

    // allExceed returns true only when ALL 4 bars are populated, each |s|≥T, and same sign.
    // Same-sign check prevents a threshold crossing + reversion from triggering a false event.
    func (b *barRing) allExceed(T float64) bool {
        var firstSign float64
        for i, ok := range b.valid {
            if !ok { return false }
            s := b.closes[i]
            if math.Abs(s) < T { return false }
            if i == 0 {
                firstSign = s
            } else if s*firstSign < 0 {
                return false
            }
        }
        return true
    }

    // computeSpread — D-06: (pxA - pxB) / mid × 10_000 bps. mid = (pxA + pxB)/2.
    func computeSpreadBps(pxLong, pxShort float64) float64 {
        mid := (pxLong + pxShort) / 2.0
        if mid <= 0 { return 0 }
        return (pxLong - pxShort) / mid * 10_000.0
    }

    // sampleLegs fetches BBOs on both legs; returns (midLong, midShort, oldestAge, err).
    // oldestAge is max(age_long, age_short) — the freshness gate uses this (D-22 PriceGapKlineStalenessSec).
    func sampleLegs(
        exchanges map[string]exchange.Exchange,
        cand models.PriceGapCandidate,
        now time.Time,
    ) (midLong, midShort float64, oldestAge time.Duration, err error) {
        longEx, ok := exchanges[cand.LongExch]
        if !ok { return 0, 0, 0, fmt.Errorf("unknown long exchange: %s", cand.LongExch) }
        shortEx, ok := exchanges[cand.ShortExch]
        if !ok { return 0, 0, 0, fmt.Errorf("unknown short exchange: %s", cand.ShortExch) }

        bboL, okL := longEx.GetBBO(cand.Symbol)
        if !okL { return 0, 0, 0, fmt.Errorf("long GetBBO not populated: %s", cand.Symbol) }
        bboS, okS := shortEx.GetBBO(cand.Symbol)
        if !okS { return 0, 0, 0, fmt.Errorf("short GetBBO not populated: %s", cand.Symbol) }

        midLong = (bboL.BidPrice + bboL.AskPrice) / 2.0
        midShort = (bboS.BidPrice + bboS.AskPrice) / 2.0

        ageL := now.Sub(bboL.UpdatedAt)
        ageS := now.Sub(bboS.UpdatedAt)
        oldestAge = ageL
        if ageS > oldestAge { oldestAge = ageS }
        return
    }

    // DetectionResult — output of detectOnce.
    type DetectionResult struct {
        Fired         bool
        SpreadBps     float64
        MidLong       float64
        MidShort      float64
        StalenessSec  float64 // oldestAge in seconds; used by risk gate for "<90s" check
        Reason        string  // populated when Fired=false, for log diagnostics
    }

    // detectOnce runs a single-tick detection for one candidate. Pure function apart from
    // BBO fetch + bar state mutation; fully unit-testable with stubExchange.
    func (t *Tracker) detectOnce(cand models.PriceGapCandidate, now time.Time) DetectionResult {
        midL, midS, age, err := sampleLegs(t.exchanges, cand, now)
        if err != nil {
            return DetectionResult{Reason: "sample_error: " + err.Error()}
        }

        spread := computeSpreadBps(midL, midS)

        t.barsMu.Lock()
        cb := t.bars[cand.ID()]
        if cb == nil {
            cb = &candidateBars{}
            t.bars[cand.ID()] = cb
        }
        cb.ring.push(now.Unix()/60, spread)
        cb.lastSampleAt = now
        fired := cb.ring.allExceed(cand.ThresholdBps) && age < time.Duration(t.cfg.PriceGapKlineStalenessSec)*time.Second
        t.barsMu.Unlock()

        reason := ""
        if !fired {
            if age >= time.Duration(t.cfg.PriceGapKlineStalenessSec)*time.Second {
                reason = "stale_bbo"
            } else {
                reason = "insufficient_persistence"
            }
        }
        return DetectionResult{
            Fired:        fired,
            SpreadBps:    spread,
            MidLong:      midL,
            MidShort:     midS,
            StalenessSec: age.Seconds(),
            Reason:       reason,
        }
    }
    ```
