    package pricegaptrader

    import (
        "context"
        "math"
        "time"

        "arb/internal/models"

    )

    // startMonitor spawns a per-position goroutine that watches for exit triggers.
    // Called from openPair (on new) and rehydrate (on startup).
    func (t *Tracker) startMonitor(pos *models.PriceGapPosition) {
        ctx, cancel := context.WithCancel(context.Background())
        t.monMu.Lock()
        t.monitors[pos.ID] = cancel
        t.monMu.Unlock()

        t.exitWG.Add(1)
        go t.monitorPosition(ctx, pos)
    }

    // monitorPosition polls at PriceGapPollIntervalSec cadence; fires exit on:
    //   (a) |spread_bps| ≤ ThresholdBps * PriceGapExitReversionFactor   [reverted]
    //   (b) time.Since(OpenedAt) ≥ PriceGapMaxHoldMin minutes            [max_hold]
    func (t *Tracker) monitorPosition(ctx context.Context, pos *models.PriceGapPosition) {
        defer t.exitWG.Done()
        defer func() {
            t.monMu.Lock(); delete(t.monitors, pos.ID); t.monMu.Unlock()
        }()

        interval := time.Duration(t.cfg.PriceGapPollIntervalSec) * time.Second
        if interval <= 0 { interval = 30 * time.Second }
        ticker := time.NewTicker(interval)
        defer ticker.Stop()

        maxHold := time.Duration(t.cfg.PriceGapMaxHoldMin) * time.Minute
        reversionFactor := t.cfg.PriceGapExitReversionFactor
        if reversionFactor <= 0 { reversionFactor = 0.5 }

        for {
            select {
            case <-ctx.Done():
                return
            case <-t.stopCh:
                return
            case now := <-ticker.C:
                // Check max-hold first — cheapest.
                if now.Sub(pos.OpenedAt) >= maxHold {
                    t.closePair(pos, models.ExitReasonMaxHold)
                    return
                }
                // Check reversion. Reuse detector sampleLegs pattern.
                cand := models.PriceGapCandidate{
                    Symbol: pos.Symbol, LongExch: pos.LongExchange,
                    ShortExch: pos.ShortExchange, ThresholdBps: pos.ThresholdBps,
                }
                midL, midS, _, err := sampleLegs(t.exchanges, cand, now)
                if err != nil {
                    t.log.Warn("pricegap: monitor sampleLegs %s: %v", pos.ID, err)
                    continue
                }
                spread := computeSpreadBps(midL, midS)
                if math.Abs(spread) <= pos.ThresholdBps*reversionFactor {
                    t.closePair(pos, models.ExitReasonReverted)
                    return
                }
            }
        }
    }

    // closePair closes both legs (simultaneous IOC; remainder via market per D-12),
    // computes realized PnL + slippage, persists with ExitReason, and moves to history.
    func (t *Tracker) closePair(pos *models.PriceGapPosition, reason string) {
        pos.Status = "exiting"
        _ = t.db.SavePriceGapPosition(pos) // best-effort state write

        longEx := t.exchanges[pos.LongExchange]
        shortEx := t.exchanges[pos.ShortExchange]
        decL := t.sizeDecimals(pos.LongExchange, pos.Symbol)
        decS := t.sizeDecimals(pos.ShortExchange, pos.Symbol)

        // Close long = sell; close short = buy.
        longResCh := make(chan fillResult, 1)
        shortResCh := make(chan fillResult, 1)
        go func() { longResCh <- t.placeCloseLeg(longEx, pos.Symbol, exchange.SideSell, pos.LongSize, decL) }()
        go func() { shortResCh <- t.placeCloseLeg(shortEx, pos.Symbol, exchange.SideBuy, pos.ShortSize, decS) }()
        lr, sr := <-longResCh, <-shortResCh

        // Retry any remainder as market (D-12).
        if lr.filled < pos.LongSize {
            rem := pos.LongSize - lr.filled
            t.closeLegMarket(longEx, pos.Symbol, exchange.SideSell, rem, decL)
            lr.filled = pos.LongSize
        }
        if sr.filled < pos.ShortSize {
            rem := pos.ShortSize - sr.filled
            t.closeLegMarket(shortEx, pos.Symbol, exchange.SideBuy, rem, decS)
            sr.filled = pos.ShortSize
        }

        // Realized PnL (delta-neutral, USDT-denominated):
        //   long: (lr.vwap - pos.LongFillPrice) * pos.LongSize
        //   short: (pos.ShortFillPrice - sr.vwap) * pos.ShortSize
        pos.RealizedPnL =
            (lr.vwap-pos.LongFillPrice)*pos.LongSize +
            (pos.ShortFillPrice-sr.vwap)*pos.ShortSize

        // Realized slippage as round-trip bps (D-21): entry asymmetry + exit asymmetry summed.
        // Entry slippage per leg (bps):
        entryLongBps  := (pos.LongFillPrice  - pos.LongMidAtDecision)  / pos.LongMidAtDecision  * 10_000
        entryShortBps := (pos.ShortMidAtDecision - pos.ShortFillPrice) / pos.ShortMidAtDecision * 10_000
        // Exit slippage approximated against current fills as mids.
        exitLongBps  := (lr.vwap - pos.LongMidAtDecision)  / pos.LongMidAtDecision  * 10_000
        exitShortBps := (pos.ShortMidAtDecision - sr.vwap) / pos.ShortMidAtDecision * 10_000
        pos.RealizedSlipBps = entryLongBps + entryShortBps + exitLongBps + exitShortBps

        pos.ExitReason = reason
        pos.Status = "closed"
        pos.ClosedAt = time.Now()
        if err := t.db.SavePriceGapPosition(pos); err != nil {
            t.log.Warn("pricegap: SavePriceGapPosition(closed) failed for %s: %v", pos.ID, err)
        }
        _ = t.db.RemoveActivePriceGapPosition(pos.ID)
        _ = t.db.AddPriceGapHistory(pos)

        // Exec-quality follow-up (PG-RISK-03) — Plan-local function in slippage.go (Task 2).
        t.recordSlippageAndMaybeDisable(pos)

        t.log.Info("pricegap: closed %s reason=%s pnl=%.4f slipBps=%.1f",
            pos.ID, reason, pos.RealizedPnL, pos.RealizedSlipBps)
    }

    // placeCloseLeg — IOC ReduceOnly exit attempt (D-12). Returns fillResult.
    func (t *Tracker) placeCloseLeg(ex exchange.Exchange, symbol string, side exchange.Side, size float64, decimals int) fillResult {
        params := exchange.PlaceOrderParams{
            Symbol:     symbol,
            Side:       side,
            OrderType:  "market",
            Size:       fmtSize(roundStep(size, decimals), decimals),
            Force:      "ioc",
            ReduceOnly: true,
        }
        id, err := ex.PlaceOrder(params)
        if err != nil { return fillResult{err: err} }
        qty, vwap, err := ex.GetOrderFilledQty(symbol, id)
        return fillResult{orderID: id, filled: qty, vwap: vwap, err: err}
    }

    // fmtSize — trivial helper; replace with utils.FormatSize if present.
    func fmtSize(v float64, decimals int) string {
        return strconvFormatFloat(v, decimals)
    }

    package pricegaptrader

    import (
        "fmt"
        "time"

        "arb/internal/models"
    )

    // recordSlippageAndMaybeDisable appends the closed trade's realized/modeled
    // slippage sample and evaluates the 10-trade rolling exec-quality rule (D-19).
    //
    // Rule: after at least 10 samples, if mean(realized) > 2 * mean(modeled),
    // set pg:candidate:disabled:<symbol> so subsequent preEntry gates block
    // with ErrPriceGapCandidateDisabled until human re-enables via cmd/pg-admin.
    func (t *Tracker) recordSlippageAndMaybeDisable(pos *models.PriceGapPosition) {
        candID := pos.Symbol + "_" + pos.LongExchange + "_" + pos.ShortExchange
        sample := models.SlippageSample{
            PositionID: pos.ID,
            Realized:   pos.RealizedSlipBps,
            Modeled:    pos.ModeledSlipBps,
            Timestamp:  time.Now(),
        }
        if err := t.db.AppendSlippageSample(candID, sample); err != nil {
            t.log.Warn("pricegap: AppendSlippageSample %s: %v", candID, err)
            return
        }

        samples, err := t.db.GetSlippageWindow(candID, 10)
        if err != nil {
            t.log.Warn("pricegap: GetSlippageWindow %s: %v", candID, err)
            return
        }
        if len(samples) < 10 { return }

        var realSum, modSum float64
        for _, s := range samples {
            realSum += s.Realized
            modSum += s.Modeled
        }
        meanReal := realSum / 10.0
        meanMod := modSum / 10.0

        // 2× threshold (D-19). Guard against divide-by-zero.
        if meanMod > 0 && meanReal > 2.0*meanMod {
            reason := fmt.Sprintf("exec_quality_auto_disable: meanReal=%.1f > 2*meanMod=%.1f at %s",
                meanReal, meanMod, time.Now().UTC().Format(time.RFC3339))
            if err := t.db.SetCandidateDisabled(pos.Symbol, reason); err != nil {
                t.log.Warn("pricegap: SetCandidateDisabled %s: %v", pos.Symbol, err)
                return
            }
            t.log.Warn("pricegap: AUTO-DISABLED %s (%s)", pos.Symbol, reason)
        }
    }
    ```

    package pricegaptrader

    import (
        "time"

        "arb/internal/models"
    )

    // rehydrate runs at Start(); for each position in pg:positions:active:
    //   - Verify BOTH legs still exist on the exchange (GetPosition size > 0).
    //   - If either leg is zero-size (exchange side closed while we were down),
    //     stamp ExitReasonOrphan and move to history.
    //   - Otherwise, re-enroll in monitorPosition goroutine.
    //
    // Idempotent: safe to call multiple times (startMonitor deletes any prior
    // monitor entry for the same posID before re-registering).
    func (t *Tracker) rehydrate() {
        active, err := t.db.GetActivePriceGapPositions()
        if err != nil {
            t.log.Warn("pricegap: rehydrate GetActivePriceGapPositions failed: %v", err)
            return
        }
        t.log.Info("pricegap: rehydrating %d active positions", len(active))

        for _, pos := range active {
            if t.legsZeroOrGone(pos) {
                pos.ExitReason = models.ExitReasonOrphan
                pos.Status = "closed"
                pos.ClosedAt = time.Now()
                _ = t.db.SavePriceGapPosition(pos)
                _ = t.db.RemoveActivePriceGapPosition(pos.ID)
                _ = t.db.AddPriceGapHistory(pos)
                t.log.Warn("pricegap: rehydrate orphaned %s — marked closed", pos.ID)
                continue
            }
            t.startMonitor(pos)
        }
    }

    // legsZeroOrGone returns true if either leg's on-exchange position size is 0
    // (exchange already closed it while tracker was offline). Uses GetPosition —
    // verify exact adapter method name via gopls at impl time.
    func (t *Tracker) legsZeroOrGone(pos *models.PriceGapPosition) bool {
        longEx, ok := t.exchanges[pos.LongExchange]
        if !ok { return true }
        shortEx, ok := t.exchanges[pos.ShortExchange]
        if !ok { return true }

        longPos, err := longEx.GetPosition(pos.Symbol)
        if err != nil { return true }
        shortPos, err := shortEx.GetPosition(pos.Symbol)
        if err != nil { return true }

        // Adjust the zero-size check to match Position struct — e.g., longPos.Size or longPos.PositionAmt.
        // Executor verifies via pkg/exchange/types.go.
        return longPos.Size == 0 || shortPos.Size == 0
    }
    ```
