```go
// Package pricegaptrader — Phase 12 ActivePositionChecker (D-05).
//
// Reads pg:positions:active SMEMBERS via *database.Client and matches the
// (Symbol, LongExch, ShortExch, Direction) tuple against active positions.
// Returns true → demote MUST be skipped (D-05); the controller holds the
// demote-streak at 6 so the demote fires on the next cycle once the active
// position closes.
package pricegaptrader

import (
    "github.com/yourorg/arb/internal/database"
    "github.com/yourorg/arb/internal/models"
)

type DBActivePositionChecker struct {
    db *database.Client
}

func NewDBActivePositionChecker(db *database.Client) *DBActivePositionChecker {
    return &DBActivePositionChecker{db: db}
}

func (g *DBActivePositionChecker) IsActiveForCandidate(c models.PriceGapCandidate) (bool, error) {
    positions, err := g.db.GetActivePriceGapPositions()
    if err != nil {
        // Fail-safe (D-05 is "skip demote when blocked"): on read error, treat
        // as blocked so we do not delete a candidate that may have an active
        // position. Demote-streak holds at 6; controller retries next cycle.
        return true, err
    }
    for _, p := range positions {
        // NOTE: read internal/models for actual field names. If Direction is
        // not yet on PriceGapPosition (Phase 999.1 added it to candidates;
        // verify position struct), fall back to comparing only Symbol +
        // LongExch + ShortExch and treat any direction match as block.
        if p.Symbol == c.Symbol &&
            p.LongExch == c.LongExch &&
            p.ShortExch == c.ShortExch &&
            p.Direction == c.Direction {
            return true, nil
        }
    }
    return false, nil
}
```
