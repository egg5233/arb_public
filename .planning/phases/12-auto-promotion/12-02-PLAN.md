// Package pricegaptrader — Phase 12 PromoteEventSink (D-10, D-11).
//
// Emits each PromoteEvent to two surfaces:
//   1. Redis LIST pg:promote:events (RPush + LTrim 1000) — REST seed source
//   2. WS hub Broadcast("pg_promote_event", payload) — live dashboard push
//
// Mirrors the pg:scan:cycles pattern from telemetry.go (RPush + LTrim 1000).
package pricegaptrader

import (
    "context"
    "encoding/json"

    "github.com/yourorg/arb/internal/database"
)

// WSBroadcaster is satisfied by *api.Hub. Defined here as a tiny interface
// so this package does not import internal/api (module-boundary rule).
type WSBroadcaster interface {
    Broadcast(eventType string, payload interface{})
}

const (
    keyPromoteEvents     = "pg:promote:events"
    promoteEventsMaxLen  = 1000
    wsEventPromoteEvent  = "pg_promote_event"
)

type RedisWSPromoteSink struct {
    db   *database.Client
    hub  WSBroadcaster
}

func NewRedisWSPromoteSink(db *database.Client, hub WSBroadcaster) *RedisWSPromoteSink {
    return &RedisWSPromoteSink{db: db, hub: hub}
}

func (s *RedisWSPromoteSink) Emit(ctx context.Context, ev PromoteEvent) error {
    raw, err := json.Marshal(ev)
    if err != nil {
        return err
    }
    rdb := s.db.GetRedis()
    if err := rdb.RPush(ctx, keyPromoteEvents, raw).Err(); err != nil {
        return err
    }
    // Bound to last 1000 (D-10).
    _ = rdb.LTrim(ctx, keyPromoteEvents, -promoteEventsMaxLen, -1).Err()
    if s.hub != nil {
        s.hub.Broadcast(wsEventPromoteEvent, ev)
    }
    return nil
}
```

// NotifyPromoteEvent fires a Telegram critical alert for a Phase 12
// PromotionController promote/demote event (D-13, D-14).
//
// Cooldown key (D-13): "pg_promote:{action}:{symbol}:{longExch}:{shortExch}:{direction}".
// Distinct events get distinct keys → existing 5-min cooldown does NOT
// suppress legitimate sequential events for different candidates.
// Same-candidate flap within 5 minutes IS throttled (intentional — the
// dashboard timeline carries flap detail).
//
// Safe to call on a nil receiver (project nil-safe pattern).
func (t *TelegramNotifier) NotifyPromoteEvent(action, symbol, longExch, shortExch, direction string, score, streak int) {
    if t == nil {
        return
    }
    key := "pg_promote:" + action + ":" + symbol + ":" + longExch + ":" + shortExch + ":" + direction
    if !t.checkCooldownAt(key, time.Now()) {
        return
    }
    t.Send("[PG %s] %s %s↔%s (%s) score=%d streak=%d",
        action, symbol, longExch, shortExch, direction, score, streak)
}

(2) ADD handler to internal/api/pricegap_discovery_handlers.go (alongside existing handlePgDiscoveryState):

// handlePgDiscoveryPromoteEvents — GET /api/pg/discovery/promote-events.
//
// Returns the pg:promote:events LIST contents (D-12 REST seed). Server
// returns NEWEST-FIRST (reverses Redis FIFO order before JSON encoding) so
// the dashboard timeline can render top-down without an additional sort.
//
// Response envelope: { ok: true, data: []PromoteEvent }
//
// Auth: Bearer token (matches handlePgDiscoveryState).
func (s *Server) handlePgDiscoveryPromoteEvents(w http.ResponseWriter, r *http.Request) {
    if !s.requireAuth(w, r) {
        return
    }
    rdb := s.db.GetRedis()
    raws, err := rdb.LRange(r.Context(), "pg:promote:events", 0, -1).Result()
    if err != nil {
        writeJSON(w, http.StatusInternalServerError, Response{Ok: false, Error: err.Error()})
        return
    }
    out := make([]pricegaptrader.PromoteEvent, 0, len(raws))
    for _, raw := range raws {
        var ev pricegaptrader.PromoteEvent
        if err := json.Unmarshal([]byte(raw), &ev); err != nil {
            // Skip malformed rather than failing the whole request.
            continue
        }
        out = append(out, ev)
    }
    // Reverse to newest-first (Redis RPush leaves oldest at index 0).
    for i, j := 0, len(out)-1; i < j; i, j = i+1, j-1 {
        out[i], out[j] = out[j], out[i]
    }
    writeJSON(w, http.StatusOK, Response{Ok: true, Data: out})
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Authenticated dashboard → REST endpoint | Bearer token verified by `s.requireAuth`; rejects 401 on missing/invalid token. |
| Redis pg:promote:events LIST → REST response | Untrusted JSON in LIST (only the controller writes, but defense-in-depth: malformed entries silently skipped instead of failing whole response). |
| Telegram bot token → Telegram API | Outbound only; existing TelegramNotifier handles secret. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-12-07 | Tampering | malformed JSON in pg:promote:events | mitigate | handlePgDiscoveryPromoteEvents skips rows that fail json.Unmarshal rather than 500-ing the whole request |
| T-12-08 | Information Disclosure | unauthenticated read of promote events | mitigate | requireAuth Bearer token check before LRANGE — 401 on miss |
| T-12-09 | DoS | Telegram spam from same-candidate flap | mitigate | per-event cooldown key (D-13) keeps 5-min throttle on same tuple while letting distinct tuples through |
| T-12-10 | DoS | unbounded Redis LIST growth | mitigate | LTrim 1000 on every Emit (RedisWSPromoteSink) |
| T-12-11 | DoS | unbounded HASH growth (cap_full_skips:{symbol}) | accept | one HASH field per symbol; bounded by PriceGapDiscoveryUniverse (≤20 symbols); operator can manually clear via `redis-cli HDEL pg:scan:metrics cap_full_skips:*` |
| T-12-12 | Repudiation | events lost if Redis down | accept | Telegram alert + Registry audit trail (config.json.bak) provide redundant audit; Redis recovery brings new events in (no historical replay needed) |

**Severity:** All medium or lower. No high blockers.

**Rollback:** Same as Plan 01 — set `cfg.PriceGapDiscoveryEnabled=false`. Additionally, `redis-cli DEL pg:promote:events` clears the timeline without affecting registry state.
