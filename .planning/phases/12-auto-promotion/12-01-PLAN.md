```go
func (p *PromotionController) Apply(ctx context.Context, summary CycleSummary) error {
    if !p.enabled() {
        return nil
    }
    p.mu.Lock()
    defer p.mu.Unlock()

    // Build set of accepted-above-threshold tuples in this cycle.
    accepted := make(map[string]CycleRecord, len(summary.Records))
    for _, r := range summary.Records {
        if r.Score >= p.threshold {
            accepted[candidateKey(r.Symbol, r.LongExch, r.ShortExch, r.Direction)] = r
        }
    }

    // Snapshot current registry once to avoid List+Delete race within this Apply.
    listed := p.registry.List()
    listedIdxByKey := make(map[string]int, len(listed))
    for i, c := range listed {
        listedIdxByKey[candidateKey(c.Symbol, c.LongExch, c.ShortExch, c.Direction)] = i
    }

    // ---- Promote streak update (D-01, D-02) ----
    // Touch keys: every accepted tuple OR any existing streak entry not in accepted resets.
    touched := map[string]bool{}
    for key, rec := range accepted {
        st := p.streaks[key]
        if st == nil {
            st = &streakState{}
            p.streaks[key] = st
        }
        st.promote++
        st.demote = 0
        touched[key] = true

        // Already in registry → no promote action; reset to keep counter from growing unbounded.
        if _, exists := listedIdxByKey[key]; exists {
            st.promote = 0
            continue
        }

        if st.promote >= 6 {
            c := models.PriceGapCandidate{
                Symbol:    rec.Symbol,
                LongExch:  rec.LongExch,
                ShortExch: rec.ShortExch,
                Direction: rec.Direction,
                // Other fields (Budget, etc.) inherit zero values; chokepoint validation
                // applies; see Pitfall 3 — match models.PriceGapCandidate field names.
            }
            err := p.registry.Add(ctx, "scanner-promote", c)
            switch {
            case err == nil:
                p.emitPromote(ctx, "promote", rec, st.promote, "score_threshold_met")
                st.promote = 0
            case errors.Is(err, ErrCapExceeded):
                _ = p.telemetry.IncCapFullSkip(ctx, rec.Symbol)
                // HOLD streak at 6 (D-08)
                st.promote = 6
            case errors.Is(err, ErrDuplicateCandidate):
                // Silent (D-08); HOLD at 6
                st.promote = 6
            default:
                // Unknown error — log via no-op for now, hold streak so we retry.
                st.promote = 6
            }
        }
    }
    // Reset promote-streak for keys NOT in accepted set.
    for key, st := range p.streaks {
        if !touched[key] {
            st.promote = 0
        }
    }

    // ---- Demote streak update (D-04, D-05) ----
    for key, idx := range listedIdxByKey {
        st := p.streaks[key]
        if st == nil {
            st = &streakState{}
            p.streaks[key] = st
        }
        if _, ok := accepted[key]; ok {
            st.demote = 0
            continue
        }
        st.demote++

        if st.demote >= 6 {
            c := listed[idx]
            blocked, _ := p.guard.IsActiveForCandidate(c)
            if blocked {
                // HOLD demote-streak at 6 (D-05) — no event, retry next cycle.
                st.demote = 6
                continue
            }
            if err := p.registry.Delete(ctx, "scanner-demote", idx); err != nil {
                // Hold streak so we retry; don't emit event on failure.
                st.demote = 6
                continue
            }
            // Synthesize a CycleRecord-like for the event payload using last-known
            // candidate fields and score=0 (no score this cycle = below threshold).
            p.emitPromote(ctx, "demote", CycleRecord{
                Symbol: c.Symbol, LongExch: c.LongExch, ShortExch: c.ShortExch,
                Direction: c.Direction, Score: 0,
            }, st.demote, "score_below_threshold")
            st.demote = 0
        }
    }
    return nil
}

func (p *PromotionController) emitPromote(ctx context.Context, action string, rec CycleRecord, streak int, reason string) {
    ev := PromoteEvent{
        TS:           p.nowFuncMS(),
        Action:       action,
        Symbol:       rec.Symbol,
        LongExch:     rec.LongExch,
        ShortExch:    rec.ShortExch,
```
