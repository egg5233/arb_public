    ```go
    func (e *SpotEngine) capitalForExchange(exchName string) float64 {
        // When unified capital is enabled, derive from pool
        if e.cfg.EnableUnifiedCapital && e.cfg.TotalCapitalUSDT > 0 {
            maxPos := e.cfg.SpotFuturesMaxPositions
            if maxPos <= 0 {
                maxPos = 1
            }
            derived := e.cfg.TotalCapitalUSDT / float64(maxPos) / 2.0
            mult := e.cfg.SizeMultiplier
            if mult <= 0 {
                mult = 1.0
            }
            return derived * mult
        }
        // Fallback: existing per-exchange capital logic
        if isSeparateAccount(exchName) {
            cap := e.cfg.SpotFuturesCapitalSeparate
            if cap <= 0 {
                cap = 200
            }
            return cap
        }
        cap := e.cfg.SpotFuturesCapitalUnified
        if cap <= 0 {
            cap = 500
        }
        return cap
    }
    ```

    ```go
    func (s *Server) handleGetAllocation(w http.ResponseWriter, r *http.Request) {
        if s.allocator == nil || !s.allocator.Enabled() {
            writeJSON(w, 200, Response{OK: true, Data: map[string]interface{}{
                "enabled": false,

            return

            writeJSON(w, 500, Response{OK: false, Error: "allocation summary: " + err.Error()})
    ```
