```go
func (a *Adapter) PlaceSpotMarginOrder(params exchange.SpotMarginOrderParams) (string, error) {
    // OKX does not support per-order auto-borrow flags.
    // Handle AutoBorrow by explicitly borrowing before the sell order.
    if params.AutoBorrow && params.Side == exchange.SideSell {
        baseCoin := extractBaseCoin(params.Symbol) // e.g., "BTCUSDT" -> "BTC"
        borrowAmt := params.Size
        if err := a.MarginBorrow(exchange.MarginBorrowParams{
            Coin:   baseCoin,
            Amount: borrowAmt,
        }); err != nil {
            return "", fmt.Errorf("PlaceSpotMarginOrder auto-borrow: %w", err)
        }
    }

    // Place the order with tdMode=cross (existing logic)
    // ... existing order placement code ...

    // Handle AutoRepay by explicitly repaying after the buy order fills
    if params.AutoRepay && params.Side == exchange.SideBuy {
        // After order fills, check margin balance and repay
        baseCoin := extractBaseCoin(params.Symbol)
        bal, err := a.GetMarginBalance(baseCoin)
        if err == nil && bal.Borrowed > 0 {
            repayAmt := fmt.Sprintf("%.8f", bal.Borrowed+bal.Interest)
            if repayErr := a.MarginRepay(exchange.MarginRepayParams{
                Coin:   baseCoin,
                Amount: repayAmt,
            }); repayErr != nil {
                // Log but don't fail -- the monitor will retry
                // Return the order ID (order itself succeeded)
            }
        }
    }

    return orderID, nil
}
```

**Important OKX implementation details:**
- `extractBaseCoin` helper: parse "SEIUSDT" -> "SEI" (strip "USDT" suffix). Check if this helper already exists in the OKX adapter.
- OKX's `MarginBorrow` endpoint is `/api/v5/account/spot-manual-borrow-repay` with `"type": "borrow"`.
- OKX's `MarginRepay` endpoint is the same URL with `"type": "repay"`.
- The `tdMode` for cross-margin spot orders should be `"cross"`.
- OKX spot symbol format is `SEI-USDT` (dash separator). The adapter's symbol mapper should handle this.
- For AutoRepay on BUY: the repay should happen AFTER confirming the order is filled, not immediately. But since `PlaceSpotMarginOrder` is a synchronous call that waits for the market order to fill (it's a market order), the repay can happen in the same function after order placement succeeds.
